---

## Install

```sh
$ npm i @chialab/wds-plugin-commonjs -D
$ yarn add @chialab/wds-plugin-commonjs -D
```

## Usage

```js
import { startDevServer } from '@web/dev-server';
import { commonjsPlugin } from '@chialab/wds-plugin-commonjs';

await startDevServer({
    plugins: [
        commonjsPlugin(),
    ],
});
```
