## Install

```sh
$ npm i @chialab/wds-plugin-hmr-css -D
$ yarn add @chialab/wds-plugin-hmr-css -D
```

## Usage

```js
import { startDevServer } from '@web/dev-server';
import { hmrCssPlugin } from '@chialab/wds-plugin-hmr-css';

await startDevServer({
    plugins: [
        hmrCssPlugin(),
    ],
});
```
